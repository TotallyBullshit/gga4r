  def mutation(g)
    @logger.debug "Mutation " + g.size.to_s + " chromosomes." if @logger  
    new_generation = g.dup
    new_generation.each do |chromosome|
      if rand > (1 - @p_mutation)
        @logger.debug "Mutate" if @logger
        chromosome.mutate 
      end
    end
  end
