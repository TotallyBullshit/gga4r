  def evaluation(g)
    @logger.debug "Evaluation " + g.size.to_s + " chromosomes." if @logger
    i = 0
    g.collect do |chromosome|
      i += 1
      @logger.debug "Evaluating chromosome #{i}:" if @logger
      @logger.debug "#{chromosome.stats.join("\n")}" if @logger
      chromosome.fitness
      chromosome
    end
  end
